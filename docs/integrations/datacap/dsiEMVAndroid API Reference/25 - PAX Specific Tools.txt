PAX Specific Tools
DAL Library
The DAL library allows access to certain device functions, such as LED Control and the ability to hide and disable the status bar or navigation bar.
Download the DAL Library
Example Usages
Showing/Hiding Status and Navigation Bars
* java
IDAL mDal = null;
ISystem mSystem = null;


try{
    mDal = DALManager.getInstance().getDAL(getApplicationContext());
} catch (Exception ex){


}
if (mDal != null){
    mSystem = mDal.getSystem();
    mSystem.setStatusBarDisable(true);
    mSystem.setStatusBarInvisible(true);
    mSystem.setNavigationBarInvisible(true);
}

public enum EBeepMode
{
    FREQUENCE_LEVEL_0((byte) 0x00),
    FREQUENCE_LEVEL_1((byte) 0x01),
    FREQUENCE_LEVEL_2((byte) 0x02),
    FREQUENCE_LEVEL_3((byte) 0x03),
    FREQUENCE_LEVEL_4((byte) 0x04),
    FREQUENCE_LEVEL_5((byte) 0x05),
    FREQUENCE_LEVEL_6((byte) 0x06);


    private byte beepMode;


    private EBeepMode(byte beepMode)
    {
        this.beepMode = beepMode;
    }


    public byte getBeepMode()
    {
        return beepMode;
    }
}

public enum ETermInfoKey
{
    SN(0),  // Terminal Serial Number
    MODEL(1),  // Terminal Model
    SP_VER(2),  // Monitor Version
    AP_VER(3),  // Android Firmware Version
    BIOS_VER(4),  // Boot Version (Ascending)
    MON_VER(5),  // Monitor Major Version; Monitor Minor Version
    MAINB_VER(6),  // Main PCB Hardware Version
    PORTB_CFG(7),  // PCB Interface Configuration Info
    EXTB_CFG(8),  // Extended PCB Configuration Info
    MAGB_CFG(9),  // MSR PCB Configuration Info
    RF_TYPE(10),  // RF Chip Type
    RPC_VER(11),  // RPC App Major Version; RPC App Minor Version
    CSN(12);  // Customer Serial Number


    private int key;


    private ETermInfoKey(int key)
    {
        this.key = key;
    }
    public int getKey()
    {
        return key;
    }
}

int ERR_CODE_CANCEL = -1;
int ERR_CODE_FINISH = 0;
/**
* Called when the scan fails, times out, or cancels
*
* @param errCode error code
* @param errMsg error message
*/

* Called when the scan is successful
*
* @param content scanning result

}
